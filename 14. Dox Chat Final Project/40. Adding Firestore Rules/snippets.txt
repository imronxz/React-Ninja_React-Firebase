    match /users/{user_id}{
      allow read, create: if request.auth != null;
      allow update: if request.auth.uid == user_id;
    }
    match /projects/{project_id}{
      allow read, create, update: if request.auth != null;
      allow delete: if request.auth.uid == project_id;
    }
